```matlab
%% slotted ALOHA
p = 0:0.001:1;
for N = [15,25,35]
    P_slot = N.*p.*(1-p).^(N-1);
    f = plot(p,P_slot);
    hold on;
end
legend('N=15','N=25','N=35')



%% pure ALOHA
p = 0:0.001:1;
for N = [15,25,35]
    P_slot = N.*p.*(1-p).^(2*(N-1));
    f = plot(p,P_slot);
    hold on;
end
legend('N=15','N=25','N=35')
```
